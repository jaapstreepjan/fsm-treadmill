43state_t FSM_EventHandler(const state_t state, const event_t event)
44{
45 state_t nextState = state;

47 // Check all transitions in the transition matrix
48 for(uint8_t i=0; i <= transition_cnt; ++i)
49 {
50 // Is the state equal to the from state?
51 if(transitions[i].from == state)
52 {
53 // And is the event equal to the event?
54 if(transitions[i].event == event)
55 {
56 // Execute the from state onExit() function
57 if(state_funcs[transitions[i].from].onExit != NULL)
58 {
59 state_funcs[transitions[i].from].onExit();
60 }
61

63 // Update for version 0.2 ORO

67
68 // Execute the to state onEntry() function
69 if(state_funcs[transitions[i].to].onEntry != NULL)
70 {
71 state_funcs[transitions[i].to].onEntry();
72 }
73
74 return nextState;
75 }
76 }
77 }
78
79 // Still here, so the event is unexpected in the current state. Remain in
80 // current state. Optionally, return the event back in the event buffer.

150void FSM_AddEvent(const event_t event)
151{
152 uint8_t tmpHead;
153
154 // Calculate index
155 tmpHead = (head + 1) & MAX_EVENTS_IN_BUFFER_MASK;
156
157 // Check if queue is full
158 if(tmpHead == tail)
159 {
160 // Queue is full, flush the event
161 return;
162 }
163
164 // Store the event in the queue
165 events[tmpHead] = event;
166
167 // Save the new index
168 head = tmpHead;
